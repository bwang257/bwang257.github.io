// Deterministic Matching
auto match(Order& bid, Order& ask) -> Trade {
  if (bid.price >= ask.price) {
    auto quantity = std::min(bid.quantity, ask.quantity);
    return Trade{bid.price, quantity};
  }
  return Trade{};
}
